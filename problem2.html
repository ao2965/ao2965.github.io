plot in manhattan by Z grade over the years.

data("rest_inspec")

american = rest_inspec |>
  filter(str_detect(cuisine_description, "American"),
         boro == "Manhattan",
         grade == "Z",
         !is.na(grade),
         !is.na(latitude),
         !is.na(longitude),
         latitude != 0,      
         longitude != 0)

plot_ly(american,
        x = ~longitude,
        y = ~latitude,
        type = "scatter",
        mode = "markers",
        color = ~grade)
american |>
  mutate(zipcode = factor(zipcode),
         zipcode = fct_reorder(zipcode, score)) |>
  plot_ly(
    x = ~zipcode,         
    y = ~score,           
    type = "box",
    color = ~zipcode,      
    colors = "viridis",
    showlegend = FALSE    
  ) |>
  layout(
    title = "Score Distribution by zipcode")
rest_inspec |>
  filter(!is.na(grade), !is.na(inspection_date)) |>
  mutate(year = lubridate::year(inspection_date)) |>
  filter(year >= 2015) |>
  count(year, grade) |>
  plot_ly(
    x = ~year,
    y = ~n,
    type = "bar",
    color = ~grade,
    colors = "viridis",
    text = ~n,
    textposition = "inside"
  ) |>
  layout(
    title = "Grade variation by Year",
    xaxis = list(title = "Year"),
    yaxis = list(title = "Number of Inspections"),
    barmode = "stack"
  )
